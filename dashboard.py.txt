import streamlit as st
import pandas as pd
import plotly.express as px
import plotly.graph_objects as go

st.set_page_config(page_title="Dashboard Excel", layout="wide")
st.title("📊 Dashboard Interativo de Planilha Excel")

uploaded_file = st.file_uploader("Faça upload da sua planilha (.xlsx)", type=["xlsx", "xls"])
if uploaded_file is not None:
    df = pd.read_excel(uploaded_file)
    st.session_state.df = df
    
    col1, col2 = st.columns(2)
    with col1:
        x_col = st.selectbox("Eixo X", df.columns)
    with col2:
        y_col = st.selectbox("Eixo Y (numérico)", [col for col in df.columns if df[col].dtype in ['int64', 'float64']])
    
    col1, col2 = st.columns(2)
    with col1:
        st.subheader("📋 Dados")
        st.dataframe(df.head(100))
    with col2:
        st.subheader("📈 Gráfico")
        fig = px.bar(df, x=x_col, y=y_col, title=f"{y_col} por {x_col}")
        st.plotly_chart(fig, use_container_width=True)
    
    st.subheader("📊 Resumo Estatístico")
    st.write(df.describe())
